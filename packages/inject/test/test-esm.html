📜 ES Module 代码

// 点击"生成代码"按钮生成

🎯 Puppeteer 示例

// Puppeteer 中使用 await page.evaluateOnNewDocument(() => { // 这里可以预先注入配置 }); await page.addScriptTag({ url: 'http://localhost:5173/index.esm.js' }); await page.evaluate(async () => { const { createInjector } = window.AIClashInject; const injector = createInjector({ provider: 'deepseek' }); await injector.inject(); // 测试能力 const state = await injector.call('thinking', 'getState'); console.log('Thinking state:', state); });

🎯 Playwright 示例

// Playwright 中使用 await page.addInitScript(() => { // 预加载脚本 }); await page.addScriptTag({ url: 'http://localhost:5173/index.esm.js' }); await page.evaluate(async () => { const { createInjector } = window.AIClashInject; const injector = createInjector({ provider: 'deepseek', adapter: 'window' }); await injector.inject(); await window.__AI_CLASH.input.fill('Hello from Playwright!'); await window.__AI_CLASH.send.send(); });
