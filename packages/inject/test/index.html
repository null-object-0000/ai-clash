🚀 快速注入代码

一键加载注入脚本并初始化 DeepSeek Provider

(async ()=>{ // 1. 加载脚本 if(!window.AIClashInject){ await new Promise((resolve,reject)=>{ const s=document.createElement('script'); s.src='http://localhost:5173/standalone.js'; s.onload=resolve; s.onerror=reject; document.head.appendChild(s); }); } // 2. 手动创建注入器（不依赖自动注入） try{ const inj=window.AIClashInject.createInjector({provider:'deepseek'}); await inj.inject(); console.log('✅ 注入完成'); console.log('📡 使用 window.__AI_CLASH 调用能力'); console.log('📌 测试：await window.__AI_CLASH.thinking.getState()'); }catch(e){ console.error('❌ 注入失败:',e); } })();

🧪 测试命令

注入完成后，在 Console 运行以下命令测试功能：

// 获取思考模式状态 await window.__AI_CLASH.thinking.getState() // 填充输入框 await window.__AI_CLASH.input.fill('你好 AI') // 发送消息 await window.__AI_CLASH.send.send() // 开始新对话 await window.__AI_CLASH.newChat.start() // 开启思考模式 await window.__AI_CLASH.thinking.sync(true) // 关闭思考模式 await window.__AI_CLASH.thinking.sync(false)
